Пройти курсы автоматизаторов.
cd <Название папки>
ls
touch - создание файла
git diff - инфо об изменениях
git log - инфа по логам
git branch <Название ветки> - создать новую ветку
git checkout <Название ветки> - переход в новую ветку
git push -u origin <Название ветки> - связать ветку с главной
git merge <Название ветки> - перенос данных из ветки в главную (main)
git branch -m <Старое название ветки> <Новое название ветки> - переименовать ветку
git push origin : <Название ветки> - удалить ветку из репозитория
git branch -d - удалить ветку с локального компа
